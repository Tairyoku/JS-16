function strToArr(str, num) {
  let a = [];
  let arr = str.match(/[а-я]+/ig);
for (let i = 0; i < arr.length ; i++) {
    if (arr[i].length > num) {
   a.push(arr[i]);
  }
}
    return a;
}
alert(strToArr('Над городом облака, коридоры, берега, проспект, река.', 7));
